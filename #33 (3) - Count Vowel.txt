#include <string>
#include <iostream>
#include <string>
using namespace std;

string ReadString() {
	string text;
	cout << "Please Enter Your Text?\n";
	getline(cin, text);
	return text;
}

bool IsVowel(char ch)
{
	ch = tolower(ch);
	return ((ch == 'a') || (ch == 'e') || (ch == 'i') || (ch == 'o') || (ch == 'u'));
}

int CountVowels(string text)
{
	int count = 0;
	for (int i = 0; i < text.length(); i++)
	{
		if (IsVowel(text[i]))
		{
			count++;
		}
	}
	return count;
}

int main() {
	string Character = ReadString();
	cout << "Number of Vowel is : " << CountVowels(Character) << endl;


	return 0;
}

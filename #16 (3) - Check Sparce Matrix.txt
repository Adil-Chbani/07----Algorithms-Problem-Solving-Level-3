#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

void PrintMatrix(int arr[3][3], int Rows, int Cols) {
    for (int i = 0; i < Rows; i++) {
        for (int j = 0; j < Cols; j++) {
            cout << setw(3) << arr[i][j] << " ";
        }
        cout << "\n";
    }
}

int CountNumber(int Matrix[3][3], int Rows, int Cols, int n) {
    int count = 0;
    for (int i = 0; i < Rows; i++) {
        for (int j = 0; j < Cols; j++) {
            if (Matrix[i][j] == n) count++;
        }
    }
    return count;
}

bool IsSparseMatrix(int Matrix[3][3], int Rows, int Cols) {
    int totalElements = Rows * Cols;
    return CountNumber(Matrix, Rows, Cols, 0) >= ceil(totalElements / 2.0);
}

int main() {
    int Matrix[3][3] = { {5,5,1}, {0,2,0}, {0,2,0} };

    cout << "\nMatrix:\n";
    PrintMatrix(Matrix, 3, 3);

    if (IsSparseMatrix(Matrix, 3, 3))
        cout << "\nThe matrix is sparse.\n";
    else
        cout << "\nThe matrix is not sparse.\n";

    return 0;
}

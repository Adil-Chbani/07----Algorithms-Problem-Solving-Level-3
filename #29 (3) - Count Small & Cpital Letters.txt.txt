#include <string>
#include <iostream>
#include <string>
using namespace std;

string ReadText() {
	string text;
	cout << "Please Enter Your Text?\n";
	 getline(cin, text);
	return text;
}


int CountCapitalLetters(string S1)
{
	int count = 0;
	for (int i = 0; i < S1.length(); i++)
	{
		if (isupper(S1[i]))
		{
			count++;
		}
	}
	return count;
}

int CountSmallLetters(string S1)
{
	int count = 0;
	for (int i = 0; i < S1.length(); i++)
	{
		if (islower(S1[i]))
		{
			count++;
		}
	}
	return count;
}

int main() {
	string Str = ReadText();

	cout << "\nString Length = " << Str.length() << endl;
	cout << "Capital Letters count = " << CountCapitalLetters(Str) << endl;
	cout << "Capital Letters count = " << CountSmallLetters(Str) << endl;

	return 0;
}

string ConvertRecordToLine(Information info, string Delimiter="#//#")
{
	string stClientRecord = "";
	stClientRecord = info.NumberAccount + Delimiter;
	stClientRecord += info.PinCode + Delimiter;
	stClientRecord += info.Name + Delimiter;
	stClientRecord += info.Phone + Delimiter;
	stClientRecord += to_string(info.AccountBalance);
	return stClientRecord;
}

#include <iostream>
#include <string>
#include <vector>
using namespace std;
struct Information
{
	string NumberAccount;
	string Name;
	string PinCode;
	string Phone;
	string AccountBalance;
};
vector<string> SplitString(string S1, string Delim)
{
	vector<string> vString;
	short pos = 0;
	string sWord;

	while ((pos = S1.find(Delim)) != string::npos)
	{
		sWord = S1.substr(0, pos);
		if (sWord != "")
		{
			vString.push_back(sWord);
		}

		S1.erase(0, pos + Delim.length());
	}

	if (S1 != "")
	{
		vString.push_back(S1);
	}

	return vString;
}

Information ConvertLinetoRecord(string S1)
{
	vector<string> vString = SplitString(S1,"#//#");
	Information info;
	info.NumberAccount  = vString[0];
	info.PinCode        = vString[1];
	info.Name           = vString[2];
	info.Phone          = vString[3];
	info.AccountBalance = vString[4];

	return info;
}

void PrintClientRecord(Information info)
{
	cout<< "Enter Account Number ? : " <<info.NumberAccount <<endl;
	cout<< "Enter PinCode ?        : " <<info.PinCode 		<<endl;
	cout<< "Enter Name ?           : " <<info.Name        	<<endl;
	cout<< "Enter Phone ?          : " <<info.Phone         <<endl;
	cout<<"Enter AccountBalance ? : " <<info.AccountBalance<<endl; 

}

int main()
{
	string stLine = "A150#//#1234#//#Adil Chbani#//#079999#//#5270.000000";
	PrintClientRecord(ConvertLinetoRecord(stLine));
}

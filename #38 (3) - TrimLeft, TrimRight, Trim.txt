#include <iostream>
#include <string>
#include <vector>
using namespace std;

string ReadString()
{
	string text;
	cout << "Enter Your String ?? \n";
	getline(cin,text);
	return text;
}

string TrimLeft(string S1)
{
	for (int i = 0; i < S1.size() - 1; i++)
	{
		if (S1[i] != ' ')
		{
			return (S1.substr(i, S1.size() - 1));
		} 
	}
}
string TrimRight(string& S1)
{
	for (int i = S1.size() - 1; i > 0 ; i--)
	{
		if (S1[i] != ' ')
		{
			return (S1.substr(0,i));
		}
	}
}
string Trim(string S1)
{

	return (TrimLeft(TrimRight(S1)));
	
}
int main()
{
	string TextString = ReadString();
	cout<< "\nTrim Left   :" << TrimLeft (TextString) << endl ;
	cout << "\nTrim Right :" << TrimRight(TextString) << endl;
	cout << "\nTrim       :" << Trim     (TextString) << endl;

}
